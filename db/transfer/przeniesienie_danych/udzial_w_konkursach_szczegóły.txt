select cp.uid
, c.cid
, c.title
, MIN(FROM_UNIXTIME(c.start)) as contest_start 
, MIN(FROM_UNIXTIME(cp.joined)) as user_joined
, MAX(FROM_UNIXTIME(cp.changed)) as user_changed 

from contest_participant as cp
inner join contest as c on (cp.cid = c.cid)
inner join users as u on (cp.uid = u.uid)

/*where FROM_UNIXTIME(c.start) >= '2017-10-01' */
where (u.status = 1)
and (cp.uid = 21127)
and (c.game_mode = 0)
GROUP BY  cp.uid
, c.cid
, c.title
